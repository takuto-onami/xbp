Arduino、モーター、センサー、NeoPixelを使って扇風機を作る

使用したコード


#include <Adafruit_NeoPixel.h>

const int DIR  = 8;
const int STEP = 9;


#define LED_PIN     6
#define NUMPIXELS   5

const int CDS_PIN = A1;


int threshold = 300;

Adafruit_NeoPixel pixels(NUMPIXELS, LED_PIN, NEO_GRB + NEO_KHZ800);

void setup() {

  pinMode(DIR, OUTPUT);
  pinMode(STEP, OUTPUT);


  pixels.begin();
  pixels.setBrightness(50);


  Serial.begin(9600);
}

void loop() {
  int val = analogRead(CDS_PIN);
  Serial.println(val);

  if (val < threshold) {

    pixels.clear();
    pixels.show();

    for (int i = 0; i < 360; i++) {
      clockwise(800);
    }

  } else {

    digitalWrite(STEP, LOW);

    pixels.clear();
    for (int i = 0; i < NUMPIXELS; i++) {
      pixels.setPixelColor(i, pixels.Color(0, 0, 255)); 
    }
    pixels.show();
  }

  delay(0);
}


void clockwise(int delaytime) {
  digitalWrite(DIR, HIGH);
  digitalWrite(STEP, HIGH);
  delayMicroseconds(delaytime);
  digitalWrite(STEP, LOW);
  delayMicroseconds(delaytime);
}

簡単なコード説明・作品紹介

CdSセンサー上で300以下の数値を出したときはモーターが動き、
300より大きい数値を出したときはLEDが光るコードを作った。
この仕組みは、昼間は扇風機が作動し、夜間はLEDが照明として機能する状況を想定したものです。
